template<class T>
struct yenxo::Default< T >

A default value tag.

This directive can be used to provide a default value for a struct member. Providing a default value the member implicitly becomes optional.

DEFINE_STRUCT(MyStruct
(string, name, Default("Siri"))
);
DEFINE_STRUCT
